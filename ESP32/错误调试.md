```c
#include <Arduino.h>       // 引入 Arduino 核心库
#include <driver/i2s.h>    // 引入 ESP32 I2S 驱动库，用于音频数据传输

// ==========================================
//               硬件引脚定义
// ==========================================

// --- 麦克风 INMP441 (I2S 输入) ---
#define MIC_I2S_WS   4     // 字选择引脚 (Word Select / LR Clock)
#define MIC_I2S_SCK  5     // 串行时钟引脚 (Serial Clock / Bit Clock)
#define MIC_I2S_SD   6     // 串行数据引脚 (Serial Data)

// --- 功放 MAX98357A (I2S 输出) ---
#define AMP_I2S_LRC  16    // 左右声道时钟 (Left/Right Clock)
#define AMP_I2S_BCLK 15    // 位时钟 (Bit Clock)
#define AMP_I2S_DIN  7     // 数据输入引脚 (Data In)

// --- 音量控制按键 ---
// 注意：GPIO 39 (VN) 仅限输入，无内部上拉电阻，必须外接 10k 上拉电阻到 3.3V
#define PIN_VOL_UP   39    // 音量增加按钮 (按下连接 GND)
#define PIN_VOL_DOWN 40    // 音量减小按钮 (按下连接 GND)

// ==========================================
//               音频与系统参数
// ==========================================

#define SAMPLE_RATE     16000  // 采样率：16kHz (16000次/秒)，适合人声语音
#define DMA_BUF_COUNT   8      // DMA 缓冲区数量：8个 (越多延迟越大，但越不容易卡顿)
#define DMA_BUF_LEN     128    // 每个缓冲区的长度：128个采样点
#define BUFFER_SIZE     1024   // 用户层处理缓冲区大小 (1024个采样点)

// 指针变量，指向我们将要在 PSRAM (伪静态随机存取存储器) 中分配的内存
int16_t *audio_buffer;

// ==========================================
//               音量控制逻辑
// ==========================================

int volume_level = 5;         // 当前音量档位 (范围 0-9)，默认启动时为第 5 档
float current_gain = 4.0;     // 当前实际使用的增益系数 (倍数)

// 定义 10 个档位的增益倍数表
// 0档为静音(0.0)，9档为最大(16.0倍)
const float gain_table[10] = {
    0.0f,   // Level 0: 静音
    0.5f,   // Level 1: 缩小一半
    1.0f,   // Level 2: 原始大小
    2.0f,   // Level 3: 2倍
    3.0f,   // Level 4: 3倍
    4.0f,   // Level 5: 4倍 (默认)
    6.0f,   // Level 6: 6倍
    8.0f,   // Level 7: 8倍
    12.0f,  // Level 8: 12倍
    16.0f   // Level 9: 16倍 (极大，注意啸叫)
};

// ==========================================
//               硬件初始化函数
// ==========================================

// --- 初始化麦克风 (使用 I2S_NUM_1) ---
void init_mic() {
    // 1. 配置 I2S 驱动参数
    i2s_config_t i2s_config = {
        .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_RX), // 模式：主机(Master) + 接收(RX)
        .sample_rate = SAMPLE_RATE,                          // 采样率：16000
        .bits_per_sample = I2S_BITS_PER_SAMPLE_16BIT,        // 位深：16位
        .channel_format = I2S_CHANNEL_FMT_ONLY_LEFT,         // 声道：只接收左声道 (INMP441特性)
        .communication_format = I2S_COMM_FORMAT_STAND_I2S,   // 通信格式：标准 I2S
        .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,            // 中断优先级：Level 1
        .dma_buf_count = DMA_BUF_COUNT,                      // DMA 缓冲数量
        .dma_buf_len = DMA_BUF_LEN,                          // DMA 缓冲长度
        .use_apll = false                                    // 不使用高精度 APLL 时钟 (省电/简单)
    };
    
    // 2. 配置 I2S 引脚映射
    i2s_pin_config_t pin_config = {
        .bck_io_num = MIC_I2S_SCK,       // 时钟脚
        .ws_io_num = MIC_I2S_WS,         // 字选脚
        .data_out_num = I2S_PIN_NO_CHANGE, // 麦克风不需要输出数据，设为无变化
        .data_in_num = MIC_I2S_SD        // 数据输入脚
    };

    // 3. 安装驱动并设置引脚
    i2s_driver_install(I2S_NUM_1, &i2s_config, 0, NULL);
    i2s_set_pin(I2S_NUM_1, &pin_config);
}

// --- 初始化功放 (使用 I2S_NUM_0) ---
void init_speaker() {
    // 1. 配置 I2S 驱动参数
    i2s_config_t i2s_config = {
        .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_TX), // 模式：主机(Master) + 发送(TX)
        .sample_rate = SAMPLE_RATE,                          // 采样率必须与麦克风一致
        .bits_per_sample = I2S_BITS_PER_SAMPLE_16BIT,        // 位深：16位
        .channel_format = I2S_CHANNEL_FMT_ONLY_LEFT,         // 声道：单声道/左声道
        .communication_format = I2S_COMM_FORMAT_STAND_I2S,   // 标准 I2S
        .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,            // 中断优先级
        .dma_buf_count = DMA_BUF_COUNT,                      // DMA 缓冲数量
        .dma_buf_len = DMA_BUF_LEN,                          // DMA 缓冲长度
        .use_apll = false
    };

    // 2. 配置 I2S 引脚映射
    i2s_pin_config_t pin_config = {
        .bck_io_num = AMP_I2S_BCLK,      // 时钟脚
        .ws_io_num = AMP_I2S_LRC,        // 字选脚
        .data_out_num = AMP_I2S_DIN,     // 数据输出脚 (发给功放)
        .data_in_num = I2S_PIN_NO_CHANGE // 功放不需要输入数据，设为无变化
    };

    // 3. 安装驱动并设置引脚
    i2s_driver_install(I2S_NUM_0, &i2s_config, 0, NULL);
    i2s_set_pin(I2S_NUM_0, &pin_config);
}

// ==========================================
//               核心音频任务
// ==========================================
// 该任务运行在 FreeRTOS 中，负责搬运和处理音频数据
void audio_pass_through_task(void *pvParameters) {
    size_t bytes_read = 0;      // 实际读取到的字节数
    size_t bytes_written = 0;   // 实际写入的字节数
    int32_t processed_sample = 0; // 32位临时变量，用于计算增益防止溢出

    Serial.println("[Audio] 实时透传任务启动...");

    // 任务死循环
    while (true) {
        // 1. 从麦克风读取数据
        // 参数：I2S端口, 缓冲区地址, 需读取长度, 实际读取长度指针, 等待时间(无限等待)
        i2s_read(I2S_NUM_1, audio_buffer, BUFFER_SIZE * sizeof(int16_t), &bytes_read, portMAX_DELAY);

        // 获取当前的音量增益 (读取全局变量)
        float gain = current_gain; 

        // 2. 数字信号处理 (DSP) - 音量调节
        // 循环遍历每一个采样点 (因为是16位音频，每2个字节是一个采样点，所以除以2)
        for(int i = 0; i < bytes_read / 2; i++) {
            
            // 乘法运算：将 16位 采样转为 32位 进行计算，防止结果超过 32767 导致翻转
            processed_sample = (int32_t)(audio_buffer[i] * gain);

            // --- 限幅处理 (Clipping) ---
            // 如果计算结果超过 int16 的上限 (32767)，强制设为最大值
            if (processed_sample > 32767) {
                processed_sample = 32767;
            } 
            // 如果计算结果低于 int16 的下限 (-32768)，强制设为最小值
            else if (processed_sample < -32768) {
                processed_sample = -32768;
            }

            // 将处理后的 32位 结果转回 16位 存回缓冲区
            audio_buffer[i] = (int16_t)processed_sample;
        }

        // 3. 将处理后的数据写入功放
        // 参数：I2S端口, 缓冲区地址, 需写入长度, 实际写入长度指针, 等待时间(无限等待)
        i2s_write(I2S_NUM_0, audio_buffer, bytes_read, &bytes_written, portMAX_DELAY);
    }
}

// ==========================================
//               辅助功能函数
// ==========================================

// --- 更新音量并打印详细信息 ---
// change 参数: +1 表示音量加，-1 表示音量减，0 表示仅刷新显示
void update_volume(int change) {
    int old_level = volume_level; // 记录旧音量用于比较
    volume_level += change;       // 改变音量

    // 边界检查：限制音量在 0 到 9 之间
    if (volume_level > 9) volume_level = 9;
    if (volume_level < 0) volume_level = 0;

    // 只有音量真正改变了，或者是初始化调用(change=0)，才执行打印
    if (volume_level != old_level || change == 0) {
        // 根据新档位更新全局增益系数
        current_gain = gain_table[volume_level];

        // --- 串口绘图与打印 ---
        Serial.print("当前音量: [");
        
        // 绘制进度条，例如: "#####-----"
        for (int i = 0; i <= 9; i++) {
            if (i < volume_level) Serial.print("#"); // 实心部分表示当前音量
            else Serial.print("-");                  // 空心部分表示剩余空间
        }
        
        // 打印数值信息
        Serial.print("]  等级: ");
        Serial.print(volume_level);
        Serial.print("/9  (增益: ");
        Serial.print(current_gain, 1); // 打印浮点数，保留1位小数
        Serial.println("x)");
    }
}

// ==========================================
//               Arduino Setup (初始化)
// ==========================================
void setup() {
    // 1. 初始化串口通信，用于调试输出
    Serial.begin(115200);
    delay(1000); // 等待串口稳定

    // 2. 配置按键引脚
    // 注意：GPIO 39 通常没有内部上拉，建议外接 10k 电阻上拉到 3.3V
    // 尽管写了 INPUT_PULLUP，但在某些引脚(如34-39)上可能不起作用，变成纯 INPUT
    pinMode(PIN_VOL_UP, INPUT_PULLUP);
    pinMode(PIN_VOL_DOWN, INPUT_PULLUP);

    // 3. 在 PSRAM (外部伪静态内存) 中申请缓冲区
    // 如果没有 PSRAM 的板子，请改用 malloc() 申请内部 RAM，但 1024 字节 ps_malloc 也可以改为 malloc
    audio_buffer = (int16_t *)ps_malloc(BUFFER_SIZE * sizeof(int16_t));
    
    // 检查内存申请是否成功
    if (audio_buffer == NULL) {
        Serial.println("PSRAM 内存分配失败！尝试使用内部 RAM...");
        audio_buffer = (int16_t *)malloc(BUFFER_SIZE * sizeof(int16_t));
        if (audio_buffer == NULL) {
             Serial.println("内存彻底分配失败，程序无法运行！");

// 变量用于按键消抖 (Debounce)
unsigned long last_debounce_time = 0;
const unsigned long debounce_delay = 200; // 消抖延时 200ms

void loop() {
    // 1. 读取按键状态 (因为是上拉，按下时接地，读取到 LOW 低电平)
    bool btn_up_pressed = (digitalRead(PIN_VOL_UP) == LOW);
    bool btn_down_pressed = (digitalRead(PIN_VOL_DOWN) == LOW);

    // 2. 按键处理逻辑 (带消抖)
    // 检查距离上次按键是否超过了消抖时间
    if ((millis() - last_debounce_time) > debounce_delay) {
        
        if (btn_up_pressed) {
            update_volume(1); // 增加音量
            last_debounce_time = millis(); // 更新最后按键时间
        } 
        else if (btn_down_pressed) {
            update_volume(-1); // 减小音量
            last_debounce_time = millis(); // 更新最后按键时间
        }
    }
    
    // 3. 延时让出 CPU 资源给后台任务 (Watchdog 需要)
    vTaskDelay(pdMS_TO_TICKS(50));
}
```

```c
#define DMA_BUF_COUNT   8      // DMA 缓冲区数量：8 个。数量越多，抗卡顿能力越强，但延迟会增加。
#define DMA_BUF_LEN     128    // 每个缓冲区的采样点数：128。
                               // 总缓冲区大小 = 8 * 128 * 2字节(16bit) = 2KB。
#define BUFFER_SIZE     1024   // 用户层处理缓冲区大小 (一次性从 DMA 搬运多少数据出来处理)

// 音频缓冲区指针
// 我们将在 setup() 中为其动态分配内存 (优先使用 PSRAM)
int16_t *audio_buffer; 

// ==========================================
//               音量与增益控制变量
// ==========================================

int volume_level = 5;         // 当前音量等级 (0-9)，默认开机为第 5 档
float current_gain = 4.0;     // 当前实际增益系数 (倍数)，默认 4 倍放大

// 预设的增益查询表 (Lookup Table)
// 这种非线性设计（0.5 -> 1 -> 2 -> ... -> 16）符合人耳对响度的对数感知特性
const float gain_table[10] = {
    0.0f,   // 等级 0: 静音
    0.5f,   // 等级 1: 衰减
    1.0f,   // 等级 2: 原声
    2.0f,   // 等级 3: 2倍放大
    3.0f,   // 等级 4: 3倍放大
    4.0f,   // 等级 5: 4倍放大 (默认)
    6.0f,   // 等级 6: 6倍放大
    8.0f,   // 等级 7: 8倍放大
    12.0f,  // 等级 8: 12倍放大
    16.0f   // 等级 9: 16倍放大 (注意：过大增益可能导致啸叫或破音)
};

//               I2S 硬件初始化

// --- 初始化麦克风 (I2S_NUM_1) ---
void init_mic() {
    // 1. 定义 I2S 配置结构体
    i2s_config_t i2s_config = {
        .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_RX), // 模式：主机(Master) + 接收(RX)
        .sample_rate = SAMPLE_RATE,                          // 采样率：16000
        .bits_per_sample = I2S_BITS_PER_SAMPLE_16BIT,        // 位宽：16位 (CD音质标准)
        .channel_format = I2S_CHANNEL_FMT_ONLY_LEFT,         // 声道：INMP441通常设为左声道接收
        .communication_format = I2S_COMM_FORMAT_STAND_I2S,   // 格式：标准 I2S 协议 (Philips Standard)
        .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,            // 中断优先级：默认 Level 1
        .dma_buf_count = DMA_BUF_COUNT,                      // DMA 描述符数量
        .dma_buf_len = DMA_BUF_LEN,                          // DMA 缓冲区长度
        .use_apll = false                                    // 不使用高精度 APLL 时钟 (语音应用不需要)
    };
    
    // 2. 定义引脚映射结构体
    i2s_pin_config_t pin_config = {
        .bck_io_num = MIC_I2S_SCK,       // 串行时钟脚
        .ws_io_num = MIC_I2S_WS,         // 左右声道选择脚
        .data_out_num = I2S_PIN_NO_CHANGE, // 麦克风是输入设备，不需要输出数据，设为无变化
        .data_in_num = MIC_I2S_SD        // 数据输入脚
    };

    // 3. 应用配置
    i2s_driver_install(I2S_NUM_1, &i2s_config, 0, NULL); // 安装驱动
    i2s_set_pin(I2S_NUM_1, &pin_config);                 // 绑定引脚
}

// --- 初始化功放 (I2S_NUM_0) ---
void init_speaker() {
    // 1. 定义 I2S 配置结构体
    i2s_config_t i2s_config = {
        .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_TX), // 模式：主机(Master) + 发送(TX)
        .sample_rate = SAMPLE_RATE,                          // 采样率必须与麦克风完全一致
        .bits_per_sample = I2S_BITS_PER_SAMPLE_16BIT,        // 位宽：16位
        .channel_format = I2S_CHANNEL_FMT_ONLY_LEFT,         // 声道：MAX98357A 默认也是接收左声道数据
        .communication_format = I2S_COMM_FORMAT_STAND_I2S,   // 格式：标准 I2S
        .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,
        .dma_buf_count = DMA_BUF_COUNT,
        .dma_buf_len = DMA_BUF_LEN,
        .use_apll = false
    };

    // 2. 定义引脚映射结构体
    i2s_pin_config_t pin_config = {
        .bck_io_num = AMP_I2S_BCLK,      // 串行时钟脚
        .ws_io_num = AMP_I2S_LRC,        // 左右声道选择脚
        .data_out_num = AMP_I2S_DIN,     // 数据输出脚 (ESP32 发送给功放)
        .data_in_num = I2S_PIN_NO_CHANGE // 功放是输出设备，不需要读入数据
    };

    // 3. 应用配置
    i2s_driver_install(I2S_NUM_0, &i2s_config, 0, NULL);
    i2s_set_pin(I2S_NUM_0, &pin_config);

// ==========================================
//               核心音频处理任务
// ==========================================
// 这个函数运行在 FreeRTOS 的独立线程中，死循环运行，不干扰主 Loop
void audio_pass_through_task(void *pvParameters) {
    size_t bytes_read = 0;        // 记录从麦克风实际读取了多少字节
    size_t bytes_written = 0;     // 记录向功放实际写入了多少字节
    int32_t processed_sample = 0; // 32位临时变量，用于计算增益 (防止16位乘法溢出)

    Serial.println("[Audio] 实时透传任务启动...");

    while (true) {
        // --- 1. 读取阶段 ---
        // 从 I2S_NUM_1 (麦克风) 读取数据到 buffer
        // portMAX_DELAY 表示如果缓冲区空了，就一直等待，直到有数据为止 (阻塞模式)
        i2s_read(I2S_NUM_1, audio_buffer, BUFFER_SIZE * sizeof(int16_t), &bytes_read, portMAX_DELAY);

        // 获取当前的增益值 (原子操作，多线程安全)
        float gain = current_gain; 

        // --- 2. 处理阶段 (DSP) ---
        // 遍历读取到的每一个采样点。注意：bytes_read 是字节数，int16 是2字节，所以除以2
        for(int i = 0; i < bytes_read / 2; i++) {
            
            // 核心算法：当前采样值 * 增益倍数
            // 必须强转为 int32_t，因为 int16 最大 32767，乘以 4 倍就会变成 130000，超出范围会变成负数杂音
            processed_sample = (int32_t)(audio_buffer[i] * gain);

            // 限幅处理 (Clipping)
            // 如果放大后的数据超过了 16位音频的物理极限，必须强制“削平”，否则会发生整数溢出翻转 (爆音)
            if (processed_sample > 32767) {
                processed_sample = 32767;
            } else if (processed_sample < -32768) {
                processed_sample = -32768;
            }

            // 将处理好的安全数据存回缓冲区
            audio_buffer[i] = (int16_t)processed_sample;
        }

        // --- 3. 写入阶段 ---
        // 将处理完的数据通过 I2S_NUM_0 发送给功放播放
        i2s_write(I2S_NUM_0, audio_buffer, bytes_read, &bytes_written, portMAX_DELAY);
    }
}

// --- 刷新 OLED 屏幕显示 ---
// 该函数负责将当前的音量状态绘制到屏幕上
void refresh_oled_display() {
    display.clearDisplay(); // 步骤1: 清空显存 (否则新旧内容会重叠)
    
    // --- 步骤2: 绘制第一行文字 (音量数值) ---
    display.setTextSize(1);              // 字体大小: 1 (最小标准字体)
    display.setTextColor(SSD1306_WHITE); // 字体颜色: 白色 (OLED 只有亮灭)
    display.setCursor(0, 0);             // 光标归位到 (x=0, y=0)
    display.print("Volume: ");           // 打印标签
    display.print(volume_level);         // 打印当前档位变量
    display.print(" / 9");               // 打印最大档位

    // --- 步骤3: 绘制进度条 (图形化反馈) ---
    // 画一个空心矩形框作为边框: x=0, y=12, 宽=128, 高=10
    display.drawRect(0, 12, 128, 10, SSD1306_WHITE);
    
    // 计算内部实心条的宽度
    // 使用 map 函数将音量 (0-9) 映射到 像素宽度 (0-126)
    // 减去2个像素是因为边框占了位置，要在框里面画
    int bar_width = map(volume_level, 0, 9, 0, 126);
    
    // 如果音量大于0，绘制实心矩形
    if (bar_width > 0) {
        // x=2, y=14 保证实心条在空心框的内部，不会覆盖边框
        display.fillRect(2, 14, bar_width, 6, SSD1306_WHITE);
    }

    // --- 步骤4: 绘制底部文字 (增益详情) ---
    display.setCursor(0, 24);            // 移动光标到底部
    display.print("Gain: ");
    display.print(current_gain, 1);      // 打印浮点数，保留 1 位小数
    display.print("x");

    // --- 步骤5: 推送显存 ---
    display.display(); // 这一步至关重要，不调用这个函数屏幕是不会亮的
}

// --- 更新音量控制逻辑 ---
// 参数 change: +1 (增加), -1 (减少), 0 (仅刷新显示)
void update_volume(int change) {
    int old_level = volume_level; // 备份旧值，用于对比是否有变化
    volume_level += change;       // 执行加减

    // 软件限位：防止数组越界
    if (volume_level > 9) volume_level = 9;
    if (volume_level < 0) volume_level = 0;

    // 只有当音量真正改变了，或者是在初始化(change=0)时，才执行耗时的屏幕刷新
    if (volume_level != old_level || change == 0) {
        // 更新全局增益系数
        current_gain = gain_table[volume_level];

        // 1. 串口输出调试信息
        Serial.print("音量等级: ");
        Serial.print(volume_level);
        Serial.print(" (增益: ");
        Serial.print(current_gain);
        Serial.println(")");

        // 2. 刷新 OLED 屏幕
        refresh_oled_display();
    }
}

//               Arduino Setup (初始化)
// ==========================================
void setup() {
    // 1. 初始化串口 (波特率 115200)
    Serial.begin(115200);
    delay(1000); // 稍微等待电源稳定

    // --- 2. 初始化 OLED 屏幕 ---
    // Wire.begin(SDA, SCL) 允许我们在 ESP32 上自定义 I2C 引脚
    Wire.begin(SCREEN_SDA, SCREEN_SCL);

    // 启动 SSD1306 驱动
    // SSD1306_SWITCHCAPVCC: 告诉屏幕芯片使用内部电路生成显示电压
    // SCREEN_ADDRESS: 0x3C 是设备地址
    if(!display.begin(SSD1306_SWITCHCAPVCC, SCREEN_ADDRESS)) {
        Serial.println(F("SSD1306 allocation failed (屏幕初始化失败)"));
        // 注意：即使屏幕失败，我们也不死循环，让音频功能继续运行
    } else {
        Serial.println(F("OLED Init Success!"));
        display.clearDisplay(); // 清屏
        display.display();      // 确保黑屏状态
    }

    // --- 3. 配置按键 ---
    // 使用 INPUT_PULLUP 尝试开启内部上拉
    // 如果是 GPIO 39/40，这一行代码可能无效，必须依赖 PCB 上的外部电阻
    pinMode(PIN_VOL_UP, INPUT_PULLUP);
    pinMode(PIN_VOL_DOWN, INPUT_PULLUP);

    // --- 4. 内存分配 ---
    // ps_malloc 尝试从 PSRAM (外部伪静态内存) 分配
    // 如果没有 PSRAM，则回退到 malloc (内部 RAM)
    audio_buffer = (int16_t *)ps_malloc(BUFFER_SIZE * sizeof(int16_t));
    if (audio_buffer == NULL) {
        Serial.println("PSRAM 内存分配失败！尝试使用内部 RAM...");
        audio_buffer = (int16_t *)malloc(BUFFER_SIZE * sizeof(int16_t));
        // 如果内部 RAM 也分配不到，说明系统内存不足，无法运行
        if (audio_buffer == NULL) while (1);
    }

    // --- 5. 初始化音频硬件 ---
    init_mic();     // 配置麦克风 I2S
    init_speaker(); // 配置功放 I2S
    
    // --- 6. 初始状态显示 ---
    update_volume(0); // 刷新一次屏幕，显示默认音量

    // --- 7. 创建 FreeRTOS 任务 ---
    // xTaskCreatePinnedToCore: 创建一个固定在特定 CPU 核心运行的任务
    // 参数: 任务函数, 任务名, 堆栈大小(8KB), 参数, 优先级(5-高), 句柄, 核心ID(0)
    xTaskCreatePinnedToCore(audio_pass_through_task, "AudioTask", 8192, NULL, 5, NULL, 0);

    Serial.println("系统就绪。");
}

// ==========================================
//               Arduino Loop (主循环)
// ==========================================
// 主循环只负责处理低速的“人机交互”(按键检测)，音频在后台任务跑

unsigned long last_debounce_time = 0;     // 上次有效按键的时间戳
const unsigned long debounce_delay = 200; // 消抖时间阈值: 200ms

void loop() {
    // 1. 读取按键物理状态 (低电平表示按下)
    bool btn_up_pressed = (digitalRead(PIN_VOL_UP) == LOW);
    bool btn_down_pressed = (digitalRead(PIN_VOL_DOWN) == LOW);

    // 2. 软件消抖逻辑
    // 只有当当前时间 - 上次响应时间 > 200ms 时，才允许再次响应
    if ((millis() - last_debounce_time) > debounce_delay) {
        
        if (btn_up_pressed) {
            update_volume(1);              // 音量 +1
            last_debounce_time = millis(); // 更新时间戳，重置计时器
        } 
        else if (btn_down_pressed) {
            update_volume(-1);             // 音量 -1
            last_debounce_time = millis(); // 更新时间戳
        }
    }
    
    // 3. 延时让出 CPU
    // 非常重要：防止主循环占用 100% CPU 导致看门狗 (Watchdog) 复位
    vTaskDelay(pdMS_TO_TICKS(50));
}
```

```c
// PCM 音频数据是“裸数据”，需要加上这个 44 字节的头，播放器才能识别
void createWavHeader(uint8_t *header, int waveDataSize) {
 
  int sampleRate = SAMPLE_RATE;        // 16000
  int byteRate = sampleRate * 2;       // 字节率 = 16000 * 2 (16位是2字节) * 1 (单声道) = 32000
  int totalDataLen = waveDataSize + 36;// 文件总长度 = 音频长度 + 36字节(除了前8字节后的头部长度)
  
  // RIFF Chunk
  // [0-3] 固定标记 "RIFF"
  header[0] = 'R'; header[1] = 'I'; header[2] = 'F'; header[3] = 'F';
  // [4-7] 文件总长度 (小端序拆分)
  header[4] = (byte)(totalDataLen & 0xFF);
  header[5] = (byte)((totalDataLen >> 8) & 0xFF);
  header[6] = (byte)((totalDataLen >> 16) & 0xFF);
  header[7] = (byte)((totalDataLen >> 24) & 0xFF); 
  // [8-11] 固定标记 "WAVE"
  header[8] = 'W'; header[9] = 'A'; header[10] = 'V'; header[11] = 'E';

  
  // fmt Chunk
  // [12-15] 固定标记 "fmt " (注意最后有个空格)
  header[12] = 'f'; header[13] = 'm'; header[14] = 't'; header[15] = ' ';
  // [16-19] 格式块长度，PCM 格式固定为 16
  header[16] = 16; header[17] = 0; header[18] = 0; header[19] = 0;
  // [20-21] 编码格式，1 代表 PCM (无压缩)
  header[20] = 1; header[21] = 0; 
  // [22-23] 声道数，1 代表单声道
  header[22] = 1; header[23] = 0; 
  // [24-27] 采样率 (16000)，同样需要拆分成4个字节
  header[24] = (byte)(sampleRate & 0xFF);
  header[25] = (byte)((sampleRate >> 8) & 0xFF);
  header[26] = (byte)((sampleRate >> 16) & 0xFF);
  header[27] = (byte)((sampleRate >> 24) & 0xFF);
  // [28-31] 字节率 (每秒播放多少字节)，播放器用它计算缓冲
  header[28] = (byte)(byteRate & 0xFF);
  header[29] = (byte)((byteRate >> 8) & 0xFF);
  header[30] = (byte)((byteRate >> 16) & 0xFF);
  header[31] = (byte)((byteRate >> 24) & 0xFF);
  // [32-33] 块对齐 (每个采样点占几字节)，16位单声道 = 2字节
  header[32] = 2; header[33] = 0; 
  // [34-35] 位深 (Bits per sample)，16位
  header[34] = 16; header[35] = 0;

  // data Chunk
  // [36-39] 固定标记 "data"
  header[36] = 'd'; header[37] = 'a'; header[38] = 't'; header[39] = 'a';
  
  // [40-43] 纯音频数据的长度 (不包含头部)
  header[40] = (byte)(waveDataSize & 0xFF);
  header[41] = (byte)((waveDataSize >> 8) & 0xFF);
  header[42] = (byte)((waveDataSize >> 16) & 0xFF);
  header[43] = (byte)((waveDataSize >> 24) & 0xFF);
}

void AudioTask(void *pvParameters) {
  // 1. 启动时先清空所有通知，防止误触发
  ulTaskNotifyTake(pdTRUE, 0);

  while (true) {
    // === 阶段 0: 待机 ===
    showStatus("Ready", "Press to Start");
    Serial.println("[Audio] Waiting for button...");
    
    // 阻塞等待开始信号
    ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
    
    // === 阶段 1: 录音中 ===
    Serial.println("[Audio] Start Recording");
    current_rec_len = 0;
    showStatus("Recording...", "Press to Stop");
    
    size_t bytes_read;
    int chunk = 1024;
    
    // 循环录音
    // 退出条件：收到停止信号(ulTaskNotifyTake > 0) 或者 内存满了
    while((current_rec_len + chunk < record_size)) {
       // 检查是否有停止信号 (不阻塞)
       if (ulTaskNotifyTake(pdTRUE, 0) > 0) {
          Serial.println("[Audio] Stop signal received");
          break; // 收到信号，跳出录音循环
       }

       i2s_read(I2S_NUM_0, (uint8_t*)rec_buffer + current_rec_len, chunk, &bytes_read, portMAX_DELAY);
       current_rec_len += bytes_read;
    }
    
    // === 阶段 2: 处理和上传 ===
    showStatus("Processing...", String(current_rec_len) + " bytes");
    vTaskDelay(pdMS_TO_TICKS(500));

    // 本地回放
    showStatus("Playing...", "Listen...");
    size_t bytes_written;
    int sent = 0;
    while(sent < current_rec_len) {
       int toSend = (current_rec_len - sent) > chunk ? chunk : (current_rec_len - sent);
       i2s_write(I2S_NUM_1, (uint8_t*)rec_buffer + sent, toSend, &bytes_written, portMAX_DELAY);
       sent += toSend;
    }
    
    // 上传
    performUpload();
    
    // === 阶段 3: 任务复位 ===
    // 关键：上传完后，再次清空可能积累的误触信号，确保回到Ready状态是干净的
    ulTaskNotifyTake(pdTRUE, 0);
    Serial.println("[Audio] Cycle done, resetting...");
  }
}
```

#### 第一块：身份识别 (RIFF Chunk)

告诉电脑：“你好，我是一个 WAV 格式的文件。”

```C++
  // 'R', 'I', 'F', 'F'：这是文件类型的魔数(Magic Number)，标志着这是 RIFF 格式资源
  header[0] = 'R'; header[1] = 'I'; header[2] = 'F'; header[3] = 'F';
  
  // 文件总长度 (除了前8个字节以外剩下的长度)
  // 计算公式：WAV头剩余部分(36字节) + 音频数据长度
  // 这里的 >> 和 & 0xFF 操作是因为 WAV 格式要求使用“小端序”存储数字（低位在前，高位在后）
  header[4] = (byte)(totalDataLen & 0xFF);         // 最低8位
  header[5] = (byte)((totalDataLen >> 8) & 0xFF);
  header[6] = (byte)((totalDataLen >> 16) & 0xFF);
  header[7] = (byte)((totalDataLen >> 24) & 0xFF); // 最高8位
  
  // 'W', 'A', 'V', 'E'：明确指定这是 WAVE 音频文件
  header[8] = 'W'; header[9] = 'A'; header[10] = 'V'; header[11] = 'E';
```

#### 第二块：音频参数说明书 (fmt Chunk)

告诉电脑：“请按照以下规格来播放我。”

```C++
  // 'f', 'm', 't', ' '：格式块的开始标记
  header[12] = 'f'; header[13] = 'm'; header[14] = 't'; header[15] = ' ';
  
  // 格式块的长度：对于 PCM 格式，固定是 16 字节
  header[16] = 16; header[17] = 0; header[18] = 0; header[19] = 0;
  
  // 音频编码格式：1 代表 PCM (无压缩的线性脉冲编码调制)
  header[20] = 1; header[21] = 0; 
  
  // 声道数：1 代表单声道 (Mono)
  header[22] = 1; header[23] = 0; 
  
  // 采样率 (Sample Rate)：这里是 16000
  // 同样需要拆分成 4 个字节，小端序存储
  header[24] = (byte)(sampleRate & 0xFF);
  header[25] = (byte)((sampleRate >> 8) & 0xFF);
  header[26] = (byte)((sampleRate >> 16) & 0xFF);
  header[27] = (byte)((sampleRate >> 24) & 0xFF);
  
  // 字节率 (Byte Rate)：每秒钟传输多少字节
  // 计算公式：采样率 × 声道数 × (位深 / 8)
  // 16000 * 1 * (16/8) = 32000 字节/秒
  header[28] = (byte)(byteRate & 0xFF);
  header[29] = (byte)((byteRate >> 8) & 0xFF);
  header[30] = (byte)((byteRate >> 16) & 0xFF);
  header[31] = (byte)((byteRate >> 24) & 0xFF);
  
  // 块对齐 (Block Align)：每个采样点占用的字节数
  // 1 * 16 / 8 = 2 字节
  header[32] = 2; header[33] = 0; 
  
  // 位深 (Bits Per Sample)：16 位
  header[34] = 16; header[35] = 0; 
```

**`multipart/form-data`**（多部分表单数据

一个标准的上传文件数据包（HTTP Body）就像一个三明治，分为三层：

1. **头部 (Head)**：说明书。告诉服务器“下面我要发一个叫 `audio_file` 的文件，文件名是 `s3_audio.wav`，类型是音频”。
2. **中间 (Content)**：真正的货物。也就是你的 WAV 音频数据（二进制）。
3. **尾部 (Tail)**：封条。告诉服务器“东西发完了”。

而 **`boundary` (分界线)** 就是用来把这三层隔开的标志。

```c
// ================= 上传函数 =================
// 将录制好的音频数据通过 HTTP POST 上传到服务器
void performUpload() {
  // 1. 检查 WiFi 连接
  if (WiFi.status() != WL_CONNECTED) {
    showStatus("Error:", "WiFi Lost!");
    return;
  }
  
  // 2. 连接服务器
  showStatus("Connecting...", host);
  WiFiClient client;
  if (!client.connect(host, httpPort)) {
    showStatus("Error:", "Connect Fail"); // 连接失败
    return;
  }
  
  // 3. 准备上传
  showStatus("Uploading...", String(current_rec_len/1024) + " KB");

  // 构建 HTTP Multipart 表单数据
  // 定义分隔符 (Boundary)：这是用来区分不同数据的标志，必须独一无二。
  String boundary = "ESP32Boundary"; // 分隔符
    // --- 构建 Body 头部 (数据的自我介绍) ---
  // 格式规定：必须以 "--" + boundary 开头
  String bodyHead = "--" + boundary + "\r\n";
  // Content-Disposition: 告诉后端这是一个表单项(form-data)。
  // name="audio_file": 【关键】这是 Django 后端识别文件的“钥匙”。你的 views.py 里写的是 request.FILES.get('audio_file')，所以这里必须一模一样。
  // filename="s3_audio.wav": 告诉后端保存成什么文件名。
  bodyHead += "Content-Disposition: form-data; name=\"audio_file\"; filename=\"s3_audio.wav\"\r\n";
  // Content-Type: 告诉后端这是音频文件。
  // 最后的 \r\n\r\n (两个换行) 表示头信息结束，紧接着就是数据了。
  bodyHead += "Content-Type: audio/wav\r\n\r\n";
  // --- 构建 Body 尾部 (封箱胶带) ---
  // 格式规定：以换行开始，以 "--" + boundary + "--" 结束。
  // 注意最后多出来的两个 "--"，这是协议规定的“结束信号”
  String bodyTail = "\r\n--" + boundary + "--\r\n";

  // 计算总内容长度 = 头部 + WAV头(44字节) + 音频数据 + 尾部
  int wavHeaderSize = 44;
  int totalContentLen = bodyHead.length() + wavHeaderSize + current_rec_len + bodyTail.length();

  // 4. 发送 HTTP 请求头
  client.println("POST " + String(url) + " HTTP/1.1");
  client.println("Host: " + String(host));
  // 【关键】告诉服务器：我发的是 multipart 数据，分隔符是 ESP32Boundary
  client.println("Content-Type: multipart/form-data; boundary=" + boundary);
  // 告诉服务器：准备接收多少字节的数据
  client.print("Content-Length: "); client.println(totalContentLen);
  client.println(); // 空行表示请求头结束

  // 5. 发送数据体
  client.print(bodyHead); // 发送表单头
  
  uint8_t wavHeader[44];
  createWavHeader(wavHeader, current_rec_len); // 生成 WAV 头
  client.write(wavHeader, 44); // 发送 WAV 头

  // 分块发送音频数据 (防止一次发太大内存溢出)
  int chunkSize = 2048;
  int sent = 0;
  uint8_t *byteBuffer = (uint8_t *)rec_buffer; 
  while (sent < current_rec_len) {
    int toSend = (current_rec_len - sent) > chunkSize ? chunkSize : (current_rec_len - sent);
    client.write(byteBuffer + sent, toSend);
    sent += toSend;
  }
  client.print(bodyTail); // 发送表单尾部

  // 6. 等待服务器响应
  long timeout = millis();
  while (client.available() == 0) {
    if (millis() - timeout > 10000) { // 10秒超时
      showStatus("Error:", "Timeout");
      client.stop();
      return;
    }
  }
  
  client.stop(); // 断开连接
  showStatus("Success!", "Sent OK"); // 上传成功提示
  delay(1500); // 停留一下，让用户看到成功提示
}
```
